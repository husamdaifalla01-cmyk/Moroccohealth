## CONTEXT

The Home Dashboard is the **command center** for patients. After login, this is where they land. It must immediately communicate value and guide toward the primary action: ordering medications.

**Product Excellence Principles:**
- Hero section: Clear value prop, single primary CTA
- Trust signals: Show pharmacy count, delivery stats, reviews
- Reduce friction: Quick reorder, saved addresses, recent pharmacies
- Progressive disclosure: Show what matters now, hide complexity

**Generative Artist Principles:**
- Cards that float with subtle shadows
- Status indicators with semantic color and glow
- Smooth transitions between states
- Depth through layered elements

'use client';

// ============================================================
// Bottom Navigation for Mobile
// ============================================================

import Link from 'next/link';
import { usePathname } from 'next/navigation';
import { Home, Search, ShoppingBag, FileText, User } from 'lucide-react';
import { motion } from 'framer-motion';

const navItems = [
  { href: '/', icon: Home, label: 'Accueil' },
  { href: '/pharmacies', icon: Search, label: 'Pharmacies' },
  { href: '/orders/new', icon: ShoppingBag, label: 'Commander' },
  { href: '/prescriptions', icon: FileText, label: 'Ordonnances' },
  { href: '/profile', icon: User, label: 'Profil' },
];

export function BottomNav() {
  const pathname = usePathname();

  return (
    <nav className="fixed bottom-0 left-0 right-0 z-50 lg:hidden">
      {/* Glassmorphism background */}
      <div className="absolute inset-0 bg-white/80 backdrop-blur-lg border-t border-gray-200" />
      
      <div className="relative safe-bottom">
        <div className="flex items-center justify-around px-2 py-2">
          {navItems.map((item) => {
            const isActive = pathname === item.href || 
              (item.href !== '/' && pathname.startsWith(item.href));
            
            return (
              <Link
                key={item.href}
                href={item.href}
                className="relative flex flex-col items-center gap-1 px-3 py-2"
              >
                {/* Active indicator */}
                {isActive && (
                  <motion.div
                    layoutId="nav-indicator"
                    className="absolute inset-0 rounded-xl bg-primary-50"
                    transition={{ type: 'spring', stiffness: 500, damping: 30 }}
                  />
                )}
                
                <item.icon 
                  className={`relative h-5 w-5 transition-colors ${
                    isActive ? 'text-primary-600' : 'text-gray-500'
                  }`}
                />
                <span 
                  className={`relative text-xs font-medium transition-colors ${
                    isActive ? 'text-primary-600' : 'text-gray-500'
                  }`}
                >
                  {item.label}
                </span>
              </Link>
            );
          })}
        </div>
      </div>
    </nav>
  );
}

// Location Store (Zustand)

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface LocationState {
  latitude: number | null;
  longitude: number | null;
  address: string | null;
  city: string | null;
  isLoading: boolean;
  error: string | null;
  
  setLocation: (lat: number, lng: number) => void;
  setAddress: (address: string, city: string) => void;
  requestLocation: () => Promise<void>;
  clearLocation: () => void;
}

export const useLocationStore = create<LocationState>()(
  persist(
    (set, get) => ({
      latitude: null,
      longitude: null,
      address: null,
      city: null,
      isLoading: false,
      error: null,

      setLocation: (latitude, longitude) => {
        set({ latitude, longitude, error: null });
      },

      setAddress: (address, city) => {
        set({ address, city });
      },

      requestLocation: async () => {
        if (!navigator.geolocation) {
          set({ error: 'GÃ©olocalisation non supportÃ©e' });
          return;
        }

        set({ isLoading: true, error: null });

        try {
          const position = await new Promise<GeolocationPosition>((resolve, reject) => {
            navigator.geolocation.getCurrentPosition(resolve, reject, {
              enableHighAccuracy: true,
              timeout: 10000,
              maximumAge: 300000, // 5 minutes cache
            });
          });

          const { latitude, longitude } = position.coords;
          set({ latitude, longitude, isLoading: false });

          // Reverse geocode (would call API in production)
          // For now, set a placeholder
          set({ city: 'Casablanca', address: 'Position actuelle' });
        } catch (error) {
          set({ 
            error: 'Impossible d\'obtenir la position', 
            isLoading: false 
          });
        }
      },

      clearLocation: () => {
        set({
          latitude: null,
          longitude: null,
          address: null,
          city: null,
          error: null,
        });
      },
    }),
    {
      name: 'dawa-location',
      partialize: (state) => ({
        latitude: state.latitude,
        longitude: state.longitude,
        address: state.address,
        city: state.city,
      }),
    }
  )
);
